
CREATE TABLE DATE_DIM
(DATE_ID INT NOT NULL PRIMARY KEY,
 DATE_T DATE NULL UNIQUE,
 YEAR_T INT NULL,
 MONTH_T INT NULL,
 DAY_T INT NULL,
 WEEKDAY_T VARCHAR(1) NULL
);

CREATE TABLE LOCATION_DIM
(LOCATION_ID INT NOT NULL IDENTITY PRIMARY KEY,
 BOROUGH VARCHAR(100) NULL,
 ZONE_T VARCHAR(100) NULL,
 SERVICE_ZONE VARCHAR(100) NULL
);


CREATE TABLE TRIPS_FACT
(TRIP_ID INT NOT NULL IDENTITY PRIMARY KEY,
 PICKUP_DATETIME DATETIME NULL ,
 DROPOFF_DATETIME DATETIME NULL ,
 PICKUP_DATE DATE NULL FOREIGN KEY REFERENCES DATE_DIM(DATE_T),
 DROPOFF_DATE DATE NULL FOREIGN KEY REFERENCES DATE_DIM(DATE_T),
 PASSENGER_COUNT INT NULL,
 TRIP_DISTANCE DECIMAL(10,8) NULL,
 PICKUP_LOCATION_ID INT NULL FOREIGN KEY REFERENCES LOCATION_DIM(LOCATION_ID),
 DROPOFF_LOCATION_ID INT NULL FOREIGN KEY REFERENCES LOCATION_DIM(LOCATION_ID),
 FARE_AMOUNT DECIMAL(20,8) NULL,
 TIP_AMOUNT DECIMAL(20,8) NULL,
 TOTAL_AMOUNT DECIMAL(20,8) NULL,
 TRIP_SRC VARCHAR(100) NOT NULL,
 LOAD_DATE DATE NOT NULL
);
